```go
func Load(cfgPath string) Config {
    cfg := DefaultConfig()

    if err != nil {
        return DefaultConfig()
    }
    return cfg
}
```

```go
func ConfigPath() (string, error) {
    home, err := os.UserHomeDir()
    if err != nil {
        return "", err
    }
    return filepath.Join(home, ".tldt.toml"), nil
}
```
